 5#include <units/isq/si/speed.h>  // IWYU pragma: keep

 7#include <units/quantity_io.h>
 8#include <iostream>
 9
10using namespace units::isq;
11
12constexpr Speed auto avg_speed(Length auto d, Time auto t) { return d / t; }
13
14int main()
15{
16  using namespace units::isq::si::literals;
17  using namespace units::isq::si::references;
18  using namespace units::aliases::isq::si::international;
19
20  constexpr Speed auto v1 = 110 * (km / h);
21  constexpr Speed auto v2 = mi_per_h<>(70.);
22  constexpr Speed auto v3 = avg_speed(220_q_km, 2_q_h);
23  constexpr Speed auto v4 = avg_speed(si::length<si::international::mile>(140), si::time<si::hour>(2));
24#if UNITS_DOWNCAST_MODE == 0
25  constexpr Speed auto v5 = quantity_cast<si::speed<si::metre_per_second>>(v3);
26  constexpr Speed auto v6 = quantity_cast<si::dim_speed, si::metre_per_second>(v4);
27#else
28  constexpr Speed auto v5 = quantity_cast<si::speed<si::metre_per_second>>(v3);
29  constexpr Speed auto v6 = quantity_cast<si::metre_per_second>(v4);
30#endif
31  constexpr Speed auto v7 = quantity_cast<int>(v6);
32
33  std::cout << v1 << '\n';                                       // 110 km/h
34  std::cout << v2 << '\n';                                       // 70 mi/h
35  std::cout << STD_FMT::format("{}", v3) << '\n';                // 110 km/h
36  std::cout << STD_FMT::format("{:*^14}", v4) << '\n';           // ***70 mi/h****
37  std::cout << STD_FMT::format("{:%Q in %q}", v5) << '\n';       // 30.5556 in m/s
38  std::cout << STD_FMT::format("{0:%Q} in {0:%q}", v6) << '\n';  // 31.2928 in m/s
39  std::cout << STD_FMT::format("{:%Q}", v7) << '\n';             // 31
40}
