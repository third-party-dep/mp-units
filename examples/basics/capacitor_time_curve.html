 1#include <units/generic/dimensionless.h>
 2#include <units/isq/dimensions/electric_current.h>
 3#include <units/isq/si/capacitance.h>
 4#include <units/isq/si/resistance.h>
 5#include <units/isq/si/time.h>
 6#include <units/isq/si/voltage.h>
 7#include <units/math.h>  // IWYU pragma: keep
 8#include <units/quantity_io.h>
 9#include <iostream>
10
11int main()
12{
13  using namespace units::isq;
14  using namespace units::isq::si::capacitance_references;
15  using namespace units::isq::si::voltage_references;
16  using namespace units::isq::si::resistance_references;
17  using namespace units::isq::si::time_references;
18
19  std::cout << "mp-units capacitor time curve example...\n";
20  std::cout.setf(std::ios_base::fixed, std::ios_base::floatfield);
21  std::cout.precision(3);
22
23  constexpr auto C = 0.47 * uF;
24  constexpr auto V0 = 5.0 * V;
25  constexpr auto RR = 4.7 * kR;  // cannot use just 'R' here
26
27  for (auto t = 0 * ms; t <= 50 * ms; ++t) {
28    const Voltage auto Vt = V0 * units::exp(-t / (RR * C));
29
30    std::cout << "at " << t << " voltage is ";
31
32    if (Vt >= 1 * V)
33      std::cout << Vt;
34    else if (Vt >= 1 * mV)
35      std::cout << quantity_cast<si::millivolt>(Vt);
36    else if (Vt >= 1 * uV)
37      std::cout << quantity_cast<si::microvolt>(Vt);
38    else if (Vt >= 1 * nV)
39      std::cout << quantity_cast<si::nanovolt>(Vt);
40    else
41      std::cout << quantity_cast<si::picovolt>(Vt);
42    std::cout << "\n";
43  }
44}
