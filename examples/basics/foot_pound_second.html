 36// Print 'a' in its current units and print its value cast to the units in each of Args
 37template<class... Args, units::Quantity Q>
 38auto fmt_line(const Q a)
 39{
 40  return STD_FMT::format("{:22}", a) + (STD_FMT::format(",{:20}", units::quantity_cast<Args>(a)) + ...);
 41}

 43// Print the ship details in the units as defined in the Ship struct, in other si::imperial units, and in SI
 44void print_details(std::string_view description, const Ship& ship)
 45{
 46  using namespace units::isq::si::fps::references;
 47  const auto waterDensity = 62.4 * (lb / ft3);
 48  std::cout << STD_FMT::format("{}\n", description);
 49  std::cout << STD_FMT::format("{:20} : {}\n", "length",
 50                               fmt_line<si::fps::length<si::fps::yard>, si::length<si::metre>>(ship.length))
 51            << STD_FMT::format("{:20} : {}\n", "draft",
 52                               fmt_line<si::fps::length<si::fps::yard>, si::length<si::metre>>(ship.draft))
 53            << STD_FMT::format("{:20} : {}\n", "beam",
 54                               fmt_line<si::fps::length<si::fps::yard>, si::length<si::metre>>(ship.beam))
 55            << STD_FMT::format("{:20} : {}\n", "mass",
 56                               fmt_line<si::fps::mass<si::fps::long_ton>, si::mass<si::tonne>>(ship.mass))
 57            << STD_FMT::format("{:20} : {}\n", "speed",
 58                               fmt_line<si::fps::speed<si::fps::knot>, si::speed<si::kilometre_per_hour>>(ship.speed))
 59            << STD_FMT::format("{:20} : {}\n", "power",
 60                               fmt_line<si::fps::power<si::fps::horse_power>, si::power<si::kilowatt>>(ship.power))
 61            << STD_FMT::format("{:20} : {}\n", "main guns",
 62                               fmt_line<si::fps::length<si::fps::inch>, si::length<si::millimetre>>(ship.mainGuns))
 63            << STD_FMT::format("{:20} : {}\n", "fire shells weighing",
 64                               fmt_line<si::fps::mass<si::fps::long_ton>, si::mass<si::kilogram>>(ship.shellMass))
 65            << STD_FMT::format(
 66                 "{:20} : {}\n", "fire shells at",
 67                 fmt_line<si::fps::speed<si::fps::mile_per_hour>, si::speed<si::kilometre_per_hour>>(ship.shellSpeed))
 68            << STD_FMT::format("{:20} : {}\n", "volume underwater",
 69                               fmt_line<si::volume<si::cubic_metre>, si::volume<si::litre>>(ship.mass / waterDensity));
 70}
