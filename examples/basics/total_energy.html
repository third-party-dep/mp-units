14using namespace units::isq;
15
16Energy auto total_energy(Momentum auto p, Mass auto m, Speed auto c)
17{
18  return sqrt(pow<2>(p * c) + pow<2>(m * pow<2>(c)));
19}
20
21void si_example()
22{
23  using namespace units::isq::si;
24  using namespace units::isq::si::energy_references;
25
26  constexpr Speed auto c = si2019::speed_of_light<>;
27
28  std::cout << "\n*** SI units (c = " << c << ") ***\n";
29
30  const Momentum auto p = 4. * GeV / c;
31  const Mass auto m = 3. * GeV / pow<2>(c);
32  const Energy auto E = total_energy(p, m, c);
33
34  std::cout << "[in GeV]\n"
35            << "p = " << p << "\n"
36            << "m = " << m << "\n"
37            << "E = " << E << "\n";
38
39  const momentum<kilogram_metre_per_second> p_si = p;
40  const mass<kilogram> m_si = m;
41  const energy<joule> E_si = total_energy(p_si, m_si, c);
42
43  std::cout << "\n[in SI units]\n"
44            << "p = " << p_si << "\n"
45            << "m = " << m_si << "\n"
46            << "E = " << E_si << "\n";
47
48  std::cout << "\n[converted from SI units back to GeV]\n"
49            << "E = " << quantity_cast<gigaelectronvolt>(E_si) << "\n";
50}

52void natural_example()
